import requests

# Шаг 1: Создание заказа
def create_order():
    url = "https://api.scooter-rent.com/orders"
    payload = {
        "customer_name": "Оксана Кубликова",
        "items": ["Самокат 1", "Шлем"]
    }
    response = requests.post(url, json=payload)
    return response.json()["track"]

# Шаг 2: Получение заказа по треку
def get_order_by_track(track):
    url = f"https://api.scooter-rent.com/orders/{track}"
    response = requests.get(url)
    return response

# Шаг 3: Проверка кода ответа
def test_order_creation():
    track = create_order()
    response = get_order_by_track(track)
    assert response.status_code == 200, f"Ожидался код 200, но получен {response.status_code}"
    print("Тест пройден успешно!")

# Запуск теста
if __name__ == "__main__":
    test_order_creation()
